Core: Class Structure

Fundamental C# class definition illustrating properties (FirstName, LastName), a constructor, and a method (GetFullName) for basic object-oriented structure. 


using System;

public class Person
{
    public string FirstName { get; set; }
    public string LastName { get; set; }

    public Person(string firstName, string lastName)
    {
        FirstName = firstName;
        LastName = lastName;
    }

    public string GetFullName()
    {
        return $"{FirstName} {LastName}";
    }

    public static void Main(string[] args)
    {
        Person person1 = new Person("John", "Doe");
        Console.WriteLine($"Person: {person1.GetFullName()}");
    }
}

LINQ: Data Querying

Demonstrates Language-Integrated Query (LINQ) for filtering a List<int>. Selects even numbers and orders them, showcasing declarative data manipulation. 


using System;
using System.Collections.Generic;
using System.Linq;

public class LinqDataQuery
{
    public static void Main(string[] args)
    {
        List<int> numbers = new List<int> { 1, 5, 2, 8, 3, 7, 4, 6, 10, 9 };

        var evenNumbersQuery = from num in numbers
                               where num % 2 == 0
                               orderby num
                               select num;

        Console.WriteLine("Filtered Even Numbers (Ascending):");
        foreach (var number in evenNumbersQuery)
        {
            Console.Write($"{number} ");
        }
        Console.WriteLine();
    }
}

Async: Non-Blocking Ops

Utilizes async and await keywords for asynchronous operations. SimulateWorkAsync mimics a long-running task without freezing the main application thread. 


using System;
using System.Threading.Tasks;

public class AsyncOperations
{
    public static async Task SimulateWorkAsync(string taskName, int delayMilliseconds)
    {
        Console.WriteLine($"Task '{taskName}' starting (Thread: {Environment.CurrentManagedThreadId})");
        await Task.Delay(delayMilliseconds);
        Console.WriteLine($"Task '{taskName}' finished.");
    }

    public static async Task Main(string[] args)
    {
        Console.WriteLine($"Main method started (Thread: {Environment.CurrentManagedThreadId}).");
        Task taskAlpha = SimulateWorkAsync("Alpha", 2000);
        Task taskBeta = SimulateWorkAsync("Beta", 1000);
        Console.WriteLine("Asynchronous tasks initiated.");
        await Task.WhenAll(taskAlpha, taskBeta);
        Console.WriteLine("All asynchronous tasks completed. Main method ending.");
    }
}

IO: File Handling

Illustrates reading all text content from a file via File.ReadAllText. Includes essential try-catch blocks for robust error handling. 


using System;
using System.IO;

public class FileInputOutput
{
    public static void Main(string[] args)
    {
        string targetFilePath = "myDocument.txt";
        if (!File.Exists(targetFilePath))
        {
            File.WriteAllText(targetFilePath, "Welcome to the Digital Nexus!\nThis is your C# documentation hub.");
            Console.WriteLine($"Demo file '{targetFilePath}' created successfully.");
        }
        try
        {
            string fileContent = File.ReadAllText(targetFilePath);
            Console.WriteLine($"--- Content of '{targetFilePath}' ---");
            Console.WriteLine(fileContent);
            Console.WriteLine("--- End of Content ---");
        }
        catch (FileNotFoundException)
        {
            Console.Error.WriteLine($"CRITICAL ERROR: The file '{targetFilePath}' could not be found.");
        }
        catch (Exception ex)
        {
            Console.Error.WriteLine($"An unexpected I/O error occurred: {ex.Message}");
        }
    }
}
